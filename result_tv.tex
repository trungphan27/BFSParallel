\begin{tablenotes}
\small
\item $S(P) = T(1)/T(P)$ (có comm).
      $S'(P) = T_{comp}(1)/T_{comp}(P)$ (chỉ tính toán, loại bỏ comm).
\item \textbf{Nhận xét quan trọng:}
  \begin{enumerate}
    \item $S'(P)$ cho thấy phần tính toán thuần túy \emph{có} tăng tốc gần tuyến tính
          ($S'(4) \approx 2.10$, $S'(😎 \approx 2.14$ — giới hạn bởi Amdahl và
          phân bố bậc R-MAT, không phải giới hạn phần cứng).
    \item $S(P)$ suy giảm nặng vì $T_{comm}$ tăng rất nhanh theo P:
          tại P=8, $T_{comm} = 86\,766$\,ms $\gg T_{comp} = 47$\,ms.
    \item Nguyên nhân cốt lõi: \texttt{MPI\_Allreduce(dist[], MAX)} truyền
          $O(n) = O(8\text{M})$ số nguyên mỗi level (6 lần/lần chạy)
          trên shared-memory (1 máy), overhead IPC tăng theo P.
    \item \textbf{Bottleneck không phải băng thông mạng} (cluster 1 máy)
          mà là \emph{latency của collective MPI trên shared memory} khi
          kích thước message rất lớn ($n \times 4$ bytes $= 32$\,MB/lần Allreduce).
  \end{enumerate}
\end{tablenotes}
\end{table}
